{{- /* Returns all page images, regardless of where they live. Works with Hugo 0.150+ and PaperMod overrides. */ -}} {{- $images := slice -}} {{- /* 1. Featured image from .Params */ -}} {{- with .Params.cover.image }} {{- if (not $.Params.cover.relative) }} {{- $images = $images | append (dict "Permalink" (. | absURL)) -}} {{- else }} {{- $images = $images | append (dict "Permalink" (path.Join $.RelPermalink . | absURL)) -}} {{- end }} {{- end }} {{- /* 2. Images bundled in resources */ -}} {{- with .Resources.ByType "image" }} {{- range . }} {{- $images = $images | append (dict "Permalink" .Permalink) -}} {{- end }} {{- end }} {{- return $images -}}
